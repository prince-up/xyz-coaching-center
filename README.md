L1 Coaching Center Website

This is a structured coaching center website powered by Next.js and Supabase.

Setup
1) Install dependencies:
	 npm install
2) Copy .env.example to .env.local and set values.
3) Run the dev server:
	 npm run dev

Supabase Tables
Run the following SQL in Supabase SQL Editor:

create table if not exists courses (
	id bigint generated by default as identity primary key,
	title text not null,
	description text,
	price text,
	created_at timestamp with time zone default now()
);

create table if not exists inquiries (
	id bigint generated by default as identity primary key,
	name text not null,
	phone text not null,
	email text not null,
	message text not null,
	created_at timestamp with time zone default now()
);

create table if not exists track_courses (
	id bigint generated by default as identity primary key,
	track text not null,
	title text not null,
	description text,
	duration text,
	level text,
	created_at timestamp with time zone default now()
);

create table if not exists teachers (
	id bigint generated by default as identity primary key,
	track text not null,
	name text not null,
	subject text not null,
	experience text,
	created_at timestamp with time zone default now()
);

create table if not exists top_performers (
	id bigint generated by default as identity primary key,
	track text not null,
	name text not null,
	achievement text not null,
	year text,
	created_at timestamp with time zone default now()
);

create table if not exists student_enrollments (
	id bigint generated by default as identity primary key,
	user_id uuid not null,
	track_course_id bigint not null references track_courses(id) on delete cascade,
	progress integer default 0,
	next_class_at timestamp with time zone,
	next_class_label text,
	created_at timestamp with time zone default now()
);

create table if not exists upcoming_classes (
	id bigint generated by default as identity primary key,
	user_id uuid not null,
	title text not null,
	starts_at timestamp with time zone not null,
	track text,
	created_at timestamp with time zone default now()
);

create table if not exists test_attempts (
	id bigint generated by default as identity primary key,
	user_id uuid not null,
	score integer not null,
	total integer not null,
	accuracy integer not null,
	rank integer,
	created_at timestamp with time zone default now()
);

create table if not exists notifications (
	id bigint generated by default as identity primary key,
	user_id uuid not null,
	title text not null,
	detail text,
	created_at timestamp with time zone default now()
);

create table if not exists study_materials (
	id bigint generated by default as identity primary key,
	user_id uuid not null,
	title text not null,
	kind text not null,
	pages integer,
	created_at timestamp with time zone default now()
);

create table if not exists weak_topics (
	id bigint generated by default as identity primary key,
	user_id uuid not null,
	topic text not null,
	accuracy integer not null,
	created_at timestamp with time zone default now()
);

create table if not exists study_tasks (
	id bigint generated by default as identity primary key,
	user_id uuid not null,
	title text not null,
	detail text,
	created_at timestamp with time zone default now()
);

create table if not exists tests (
	id bigint generated by default as identity primary key,
	title text not null,
	track text,
	type text,
	total_questions integer not null,
	duration_minutes integer not null,
	negative_mark numeric default 1,
	marks_per_question numeric default 1,
	created_at timestamp with time zone default now()
);

create table if not exists test_questions (
	id bigint generated by default as identity primary key,
	test_id bigint not null references tests(id) on delete cascade,
	question_text text not null,
	options jsonb not null,
	correct_index integer not null,
	subject text,
	created_at timestamp with time zone default now()
);

create table if not exists lectures (
	id bigint generated by default as identity primary key,
	title text not null,
	track text,
	topic text,
	teacher text,
	video_url text,
	duration_minutes integer,
	created_at timestamp with time zone default now()
);

create table if not exists materials_library (
	id bigint generated by default as identity primary key,
	title text not null,
	track text,
	kind text not null,
	pages integer,
	file_url text,
	created_at timestamp with time zone default now()
);

create table if not exists announcements (
	id bigint generated by default as identity primary key,
	title text not null,
	detail text,
	created_at timestamp with time zone default now()
);

create table if not exists doubts (
	id bigint generated by default as identity primary key,
	user_id uuid not null,
	title text not null,
	question text not null,
	subject text,
	image_url text,
	status text default 'open',
	created_at timestamp with time zone default now()
);

create table if not exists doubt_replies (
	id bigint generated by default as identity primary key,
	doubt_id bigint not null references doubts(id) on delete cascade,
	teacher_name text,
	reply text not null,
	created_at timestamp with time zone default now()
);

alter table test_attempts add column if not exists test_id bigint;
alter table test_attempts add column if not exists correct_count integer;
alter table test_attempts add column if not exists wrong_count integer;
alter table test_attempts add column if not exists skipped_count integer;

Recommended RLS
Enable Row Level Security and use policies like:

-- Allow public to insert inquiries
create policy "public insert inquiries" on inquiries
for insert to anon, authenticated
with check (true);

-- Allow public read for track data
create policy "public read track courses" on track_courses
for select to anon, authenticated
using (true);

create policy "public read teachers" on teachers
for select to anon, authenticated
using (true);

create policy "public read top performers" on top_performers
for select to anon, authenticated
using (true);
